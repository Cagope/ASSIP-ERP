# ğŸ—“ Fecha y hora actual para el nombre del backup
$timestamp = Get-Date -Format "yyyyMMdd_HHmmss"

# ğŸ“ Carpeta de origen
$source = "D:\assip-erp"

# ğŸ“¦ Carpeta de destino para los backups
$backupDir = "D:\backups_assip"

# ğŸ§¾ Nombre del archivo ZIP con la fecha
$zipFile = "$backupDir\assip-erp_backup_$timestamp.zip"

# ğŸ›  Crear carpeta de backup si no existe
if (-not (Test-Path $backupDir)) {
    New-Item -ItemType Directory -Path $backupDir | Out-Null
}

# ğŸ§© Patrones a excluir (carpetas pesadas)
$excludePatterns = @("node_modules", "target", ".idea", ".git")

# ğŸ” Buscar todos los archivos y excluir los no deseados
$itemsToZip = Get-ChildItem -Path $source -Recurse -File | Where-Object {
    $pathParts = $_.FullName.Split('\')
    -not ($pathParts | Where-Object { $excludePatterns -contains $_ })
}

# ğŸ“¦ Crear el archivo ZIP con todos los archivos vÃ¡lidos
Compress-Archive -Path $itemsToZip.FullName -DestinationPath $zipFile -CompressionLevel Optimal -Force

Write-Host "`nâœ… Backup creado correctamente:`n$zipFile`n"
